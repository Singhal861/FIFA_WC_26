REQUIREMENT 1: TOURNAMENT BRACKET / FIXTURE GRAPHICS
================================================================================

Description:
  Display complete knockout bracket with dynamic TBD opponent logic:
  - Bracket structure: Team A vs Team B → Winner advances
                      Team C vs Team D → Winner advances
                      Now Winner(A,B) vs Winner(C,D) → Champion
  - Team flags/logos displayed with team names below
  - Click flag → Show scorers in THAT match with goal timing (23', 45+3, 90+3, 108')
  - Match times in UTC
  - Stadium name, city, country
  - TBD logic: Shows "Winner of QF1" or "TBD" for unconfirmed matchups
  - Match status: Finished/Live/Upcoming (if not finished, goals show "NA")

Gold Tables:
  1. gold_fact_match_schedule (unified match fact table)
  2. gold_team_goals (goal-by-goal details)

Key Features:
  ✅ Complete bracket structure with all knockout rounds
  ✅ TBD opponent logic ("Winner of Match X" or "TBD")
  ✅ Team logos (home & away) with team names
  ✅ Click flag → Show match-specific goal details with timing
  ✅ Goal timing format: "23'", "45+3'", "90+3'", "108'" (extra time)
  ✅ Penalty indicators: "(PEN)" suffix
  ✅ Match times in UTC
  ✅ Stadium information (name, city, country)
  ✅ Match status handling (Finished/Live/Upcoming)
  ✅ "NA" display for goals when match not completed

Dashboard SQL Query (Single Comprehensive Query):

-- Complete tournament bracket with goal details and TBD logic
WITH match_bracket AS (
    -- Get all knockout matches with bracket structure
    SELECT 
        match_id,
        stage,
        bracket_position,
        feeds_into_position,
        bracket_half,
        
        -- Home team (handles TBD logic)
        home_team_name,
        COALESCE(home_display_name, home_team_name, 'TBD') AS home_display_name,
        home_team_logo,
        home_score,
        
        -- Away team (handles TBD logic)
        away_team_name,
        COALESCE(away_display_name, away_team_name, 'TBD') AS away_display_name,
        away_team_logo,
        away_score,
        
        -- Match outcome
        winner_team_name,
        winner_team_logo,
        
        -- Match metadata
        match_status,
        match_datetime_utc,
        match_date_local,
        stadium_name,
        stadium_city,
        actual_country,
        is_finished,
        is_live
        
    FROM singhal.fifa_worldcup_gold.gold_fact_match_schedule
    WHERE is_knockout = TRUE
),

home_goals AS (
    -- Get goals scored by home team in each match
    SELECT
        g.match_id,
        COLLECT_LIST(
            STRUCT(
                g.scorer_name,
                g.minute,
                g.is_penalty,
                g.goal_number_in_match
            )
        ) AS goals
    FROM singhal.fifa_worldcup_gold.gold_team_goals g
    INNER JOIN match_bracket mb ON g.match_id = mb.match_id AND g.team_name = mb.home_team_name
    GROUP BY g.match_id
),

away_goals AS (
    -- Get goals scored by away team in each match
    SELECT
        g.match_id,
        COLLECT_LIST(
            STRUCT(
                g.scorer_name,
                g.minute,
                g.is_penalty,
                g.goal_number_in_match
            )
        ) AS goals
    FROM singhal.fifa_worldcup_gold.gold_team_goals g
    INNER JOIN match_bracket mb ON g.match_id = mb.match_id AND g.team_name = mb.away_team_name
    GROUP BY g.match_id
)

-- Final comprehensive query
SELECT
    -- Bracket structure
    mb.match_id,
    mb.stage,
    mb.bracket_position,              -- e.g., "QF1", "SF1", "FIN"
    mb.feeds_into_position,           -- e.g., QF1 → SF1
    mb.bracket_half,                  -- "Top" or "Bottom"
    
    -- Home team
    mb.home_team_name,
    mb.home_display_name,             -- Shows "Winner of QF1" or "TBD" for unfilled
    mb.home_team_logo,                -- 🇦🇷 Team flag
    mb.home_score,
    
    -- Home team goals (formatted with timing)
    CASE 
        WHEN mb.is_finished = TRUE AND hg.goals IS NOT NULL
        THEN TRANSFORM(
            hg.goals,
            g -> CONCAT(
                g.scorer_name, ' (', g.minute, ')',
                CASE WHEN g.is_penalty THEN ' (PEN)' ELSE '' END
            )
        )
        ELSE ARRAY('NA')  -- Match not completed yet
    END AS home_goals_detail,
    
    -- Away team
    mb.away_team_name,
    mb.away_display_name,             -- Shows "Winner of QF3" or "TBD" for unfilled
    mb.away_team_logo,                -- 🇫🇷 Team flag
    mb.away_score,
    
    -- Away team goals (formatted with timing)
    CASE 
        WHEN mb.is_finished = TRUE AND ag.goals IS NOT NULL
        THEN TRANSFORM(
            ag.goals,
            g -> CONCAT(
                g.scorer_name, ' (', g.minute, ')',
                CASE WHEN g.is_penalty THEN ' (PEN)' ELSE '' END
            )
        )
        ELSE ARRAY('NA')  -- Match not completed yet
    END AS away_goals_detail,
    
    -- Match outcome
    mb.winner_team_name,
    mb.winner_team_logo,
    
    -- Match info
    mb.match_status,                  -- 'Finished', 'Live', 'Upcoming'
    mb.match_datetime_utc,            -- UTC time
    mb.match_date_local,              -- Local timezone display
    mb.stadium_name,                  -- MetLife Stadium
    mb.stadium_city,                  -- New Jersey
    mb.actual_country,                -- United States
    mb.is_finished,
    mb.is_live

FROM match_bracket mb
LEFT JOIN home_goals hg ON mb.match_id = hg.match_id
LEFT JOIN away_goals ag ON mb.match_id = ag.match_id

ORDER BY 
    CASE mb.stage
        WHEN 'Round of 32' THEN 1
        WHEN 'Round of 16' THEN 2
        WHEN 'Quarter Final' THEN 3
        WHEN 'Semi Final' THEN 4
        WHEN 'Third Place' THEN 5
        WHEN 'Final' THEN 6
    END,
    mb.bracket_position;

    match_date_local,
    home_team_id,

    home_team_logo,

Dashboard SQL Query:

-- Query 4: Get all finished matches with results and penalty shootout detection
SELECT 
    m.match_id,
    
    -- Team information
    m.home_team_name AS team_a_name,
    m.home_team_logo AS team_a_logo,
    m.away_team_name AS team_b_name,
    m.away_team_logo AS team_b_logo,
    
    -- Score information
    m.home_score,
    m.away_score,
    
    -- Winner information
    m.winner_team_name,
    m.winner_team_logo,
    
    -- Penalty shootout detection
    -- If knockout match ended in a draw but has a winner, it was decided by penalties
    CASE 
        WHEN m.is_knockout = TRUE 
         AND m.home_score = m.away_score 
         AND m.winner_team_name IS NOT NULL 
        THEN TRUE 
        ELSE FALSE 
    END AS is_penalty_shootout,
    
    -- Match timing
    m.match_datetime_utc,
    m.match_date_local,
    
    -- Location
    m.stadium_city,
    m.stadium_name,
    m.actual_country,
    
    -- Tournament context
    m.stage,
    m.group_name,  -- NULL for knockout matches
    m.is_knockout
    
FROM singhal.fifa_worldcup_gold.gold_fact_match_schedule m
WHERE m.is_finished = TRUE
ORDER BY m.match_datetime_utc DESC;

Notes:
  - Shows ALL completed matches (both group stage and knockout)
  - Ordered by most recent first
  - Penalty shootout detection logic:
    * Knockout match (is_knockout = TRUE)
    * Match score is tied (home_score = away_score)
    * Has a winner (winner_team_name IS NOT NULL)
    * This indicates the match was decided by penalty shootout
  - Penalty shootout SCORES are not available in the data
  - Dashboard will format display as:
    * Regular wins: "Argentina 2-1 Brazil"
    * Penalty wins: "Argentina 3-3 France (penalties)" - without showing penalty score
  - group_name is NULL for knockout matches

  1. gold_golden_boot_race

Key Features:
  ✅ Match-by-match progression (match_sequence)
  ✅ Player images (player_logo field)
  ✅ Team logos
  ✅ Top 3 rankings at each match
  ✅ Cumulative stats (goals, assists, minutes)
  ✅ Medals (🥇🥈🥉) based on ranking
  ✅ Current top 3 flag (is_current_top_3)
  ✅ FIFA tiebreaker rules:
     1. Most goals
     2. Most assists (if tied on goals)
     3. Fewest minutes (if tied on goals & assists)

Dashboard SQL Query:

-- Query 6: Get Golden Boot race progression (single query for everything)
-- Use for line graph AND current standings (filter where is_current_top_3 = TRUE)
SELECT 
    match_sequence,  -- X-axis: Match number (1, 2, 3...)
    player_id,
    player_name,
    player_logo,  -- Player headshot for legend
    team_id,
    team_name,
    team_logo,
    goals_cumulative,  -- Y-axis: Cumulative goals
    assists_cumulative,
    minutes_cumulative,
    rank_at_match,  -- 1, 2, or 3
    medal,  -- 🥇, 🥈, or 🥉
    is_current_top_3  -- TRUE only for latest match (filter for current standings)
FROM singhal.fifa_worldcup_gold.gold_golden_boot_race
WHERE rank_at_match <= 3  -- Only top 3 players
ORDER BY match_sequence, rank_at_match;

Dashboard Usage:
  - Line Graph: Use all rows (shows progression over time)
  - Current Standings/Medals: Filter WHERE is_current_top_3 = TRUE (latest state only)

Query Output Columns:
  - match_sequence: Tournament match number (1, 2, 3...)
  - player_id, player_name, player_logo: Player identification
  - team_id, team_name, team_logo: Team identification
  - goals_cumulative: Total goals scored up to this match
  - assists_cumulative: Total assists up to this match
  - minutes_cumulative: Total minutes played up to this match
  - rank_at_match: Rank at this specific match (1, 2, or 3)
  - medal: Medal emoji (🥇, 🥈, or 🥉)
  - is_current_top_3: TRUE if this is the most recent match (for filtering)

  ✅ Tournament identification (name)
  ✅ Match progress tracking (total, completed, remaining)
  ✅ Goal statistics (total, average per match)
  ✅ Current tournament round/stage
  ✅ Teams remaining count
  ✅ Top scorer information (name, goals)

Dashboard SQL Query:

-- Query 7: Get tournament summary statistics
SELECT 
    tournament_name,        -- e.g., "FIFA World Cup 2026"
    total_matches,          -- Total matches in tournament
    completed_matches,      -- Matches finished
    remaining_matches,      -- Matches yet to be played
    total_goals,            -- Total goals scored
    avg_goals_per_match,    -- Average goals per completed match
    current_round,          -- e.g., "Group Stage", "Quarter Finals", "Semi Finals"
    teams_remaining,        -- Number of teams still in competition
    top_scorer_name,        -- Current leading scorer
    top_scorer_goals        -- Goals by top scorer
FROM singhal.fifa_worldcup_gold.gold_tournament_summary;

Query Output Columns:
  - tournament_name (string): Tournament name
  - total_matches (bigint): Total matches scheduled
  - completed_matches (bigint): Matches finished
  - remaining_matches (bigint): Matches yet to be played
  - total_goals (bigint): Total goals scored in tournament
  - avg_goals_per_match (double): Average goals per completed match
  - current_round (string): Current tournament stage
  - teams_remaining (bigint): Teams still competing
  - top_scorer_name (string): Leading goal scorer
  - top_scorer_goals (int): Goals by top scorer

Notes:
  - Single-row result (tournament-level aggregate)
  - Updates dynamically as matches complete
  - avg_goals_per_match = total_goals / completed_matches
  - remaining_matches = total_matches - completed_matches
  - top_scorer synced with gold_player_leaderboard
